# Interacting with the database; the strat

Gone with a DDD layered approach with repositories per domain.

## /domain

-   /session (defines interfaces and types shared across layers, e.g. `Session` and `SessionRepo`)

```golang
// domain/session/app_session.go
type Session struct {
    ID        uuid.UUID
    AppID     uuid.UUID
    StartedAt time.Time
    ...
}

type SessionRepo interface {
    Create(ctx context.Context, s Session) error
    GetByID(ctx context.Context, id uuid.UUID) (*Session, error)
}
```

## /infra

-   /db -> sqlc generated code + sql queries + migrations
-   /repo -> Implementation of repositories

```golang
// /infra/repo/session_repo.go
type sessionRepo struct {
    queries *db.Queries // <- generated by sqlc
    db      *sql.DB
}

func NewSessionRepo(db *sql.DB) domain.SessionRepo {
    return &sessionRepo{
        queries: db.New(db),
        db:      db,
    }
}

func (r *sessionRepo) Create(ctx context.Context, s domain.Session) error {
    return r.queries.InsertSession(ctx, db.InsertSessionParams{
        ID:        s.ID,
        AppID:     s.AppID,
        StartedAt: s.StartedAt,
        ...
    })
}
```

## /app

-   /session (business logic layer.. no DB or I/O)

```golang
// app/session/service.go
type SessionService struct {
    sessions domain.SessionRepo
}

func NewSessionService(r domain.SessionRepo) *SessionService {
    return &SessionService{sessions: r}
}

func (s *SessionService) StartNewSession(ctx context.Context, appID uuid.UUID) (*domain.Session, error) {
    sess := domain.Session{
        ID: uuid.New(),
        AppID: appID,
        StartedAt: time.Now(),
    }
    if err := s.sessions.Create(ctx, sess); err != nil {
        return nil, err
    }
    return &sess, nil
}
```
